122void ControlNivelTask(void *pvParameter) {
123 bool bomba_activada = false;
124
125 while (true) {
126 /* Esperar notificación del timer */
127 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
128
129 // Si está activo el control manual, salteamos el control automático
130 if (control_manual_activo) {
131 continue;
132 }
133
134 if (nivel_agua_cm < NIVEL_MIN_CM) {
135 // Activar bomba (rele y LED)
136 GPIOOff(GPIO_6); // ejemplo: relé en GPIO_6
137 //LedOn(LED_1); // LED indica bomba activa
138 estado_tanque = NIVEL_BAJO;
139
140 } else if (nivel_agua_cm > NIVEL_MAX_CM) {
141 // Apagar bomba (rele y LED)
142 GPIOOn(GPIO_6);
143 //LedOff(LED_1);
144 estado_tanque = TANQUE_LLENO;
145
146 } else {
147 estado_tanque = NIVEL_ESTABLE;
148 }
149
150 UartSendString(UART_PC, "Estado: ");
151
152 switch (estado_tanque) {
153 case NIVEL_BAJO:
154 UartSendString(UART_PC, "Nivel bajo, activando bomba\r\n");
155 break;
156 case NIVEL_ESTABLE:
157 UartSendString(UART_PC, "Nivel estable\r\n");
158 break;
159 case TANQUE_LLENO:
160 UartSendString(UART_PC, "Tanque lleno, desactivando bomba\r\n");
161 break;
162 }
163
164 vTaskDelay(pdMS_TO_TICKS(50)); // pequeña pausa
165 }
166}

172void MedirNivelTask(void *pvParameter) {
173 uint16_t distancia_cm;
174 uint16_t nivel_cm;
175
176 while (true) {
177 /* Espera la notificación del timer */
178 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
179
180 /* Leer distancia del sensor (cm) */
181 distancia_cm = HcSr04ReadDistanceInCentimeters();
182
183 /* Calcular nivel (altura - distancia) */
184 if (distancia_cm >= ALTURA_TANQUE_CM)
185 nivel_cm = 0;
186 else
187 nivel_cm = ALTURA_TANQUE_CM - distancia_cm;
188
189 /* Guardar el valor global */
190 nivel_agua_cm = nivel_cm;
191 /* Enviar por UART */
192 UartSendString(UART_PC, "Nivel de agua: ");
193 UartSendString(UART_PC, (char *)UartItoa((uint32_t)nivel_agua_cm, 10));
194 UartSendString(UART_PC, " cm\r\n");
195
196 //Muestra el nivel en el display LCD
197 LcdItsE0803Write(nivel_agua_cm);
198
199 // ---------- Envío para el Serial Plotter ----------
200 char mensaje[30];
201 char *valor_str = (char *)UartItoa((uint32_t)nivel_agua_cm, 10);
202 strcpy(mensaje, ">brightness:");
203 strcat(mensaje, valor_str);
204 strcat(mensaje, "\r\n");
205 UartSendString(UART_PC, mensaje);
206
207
208
209 }

217void ControlManualTask(void *pvParameter) {
218
219while (1) {
220 int8_t estado_switch = SwitchesRead();
221
222 /* Si el usuario presiona el SWITCH_1, alterna entre modo manual y automático */
223 if (estado_switch == SWITCH_1) {
224 if (control_manual_activo) {
225 // Si ya estaba en modo manual → volver a automático
226 LedOff(LED_1);
227 control_manual_activo = false;
228 UartSendString(UART_PC, "\r\n[CONTROL MANUAL] Desactivado, vuelve a control automático\r\n");
229
230 } else {
231 // Si estaba en automático → pasar a manual y apagar bomba
232 GPIOOn(GPIO_6); // apagar bomba
233 LedOn(LED_1);
234 control_manual_activo = true;
235 UartSendString(UART_PC, "\r\n[CONTROL MANUAL] Activado: bomba detenida manualmente\r\n");
236 }
237 }
238
239 /* SWITCH_2: alterna modo manual de válvula */
240 else if (estado_switch == SWITCH_2) {
241 if (control_valvula_manual) {
242 // Estaba activada manualmente → apagar válvula
243 control_valvula_manual = false;
244 GPIOOn(GPIO_7); // cerrar válvula
245 LedOff(LED_2);
246 UartSendString(UART_PC, "\r\n[CONTROL MANUAL] Válvula cerrada\r\n");
247 } else {
248 // Estaba apagada → activar manualmente
249 control_valvula_manual = true;
250 GPIOOff(GPIO_7); // abrir válvula
251 LedOn(LED_2);
252 UartSendString(UART_PC, "\r\n[CONTROL MANUAL] Válvula en desagote manual\r\n");
253 }
254 }
255 vTaskDelay(pdMS_TO_TICKS(100)); // pequeño retardo para evitar rebotes
256}

273void app_main(void) {
274 LedsInit();
275 HcSr04Init(GPIO_3, GPIO_2);
276 LcdItsE0803Init();
277 SwitchesInit();
278 //BOMBA
279 GPIOInit(GPIO_6, GPIO_OUTPUT);

283 GPIOOn(GPIO_7);
284
285 /* Configuración dela uart */
286 serial_config_t uart_cfg = {
287 .port = UART_PC,
288 .baud_rate = UART_BAUDRATE,
289 .func_p = NULL, // sin interrupciones de RX por ahora
290 .param_p = NULL
291 };
292 UartInit(&uart_cfg);
293
294 /* Configuración del timer para medicion: 100 ms */
295 timer_config_t timer_nivel = {
296 .timer = TIMER_A,
297 .period = REFRESH_PERIOD_US,
298 .func_p = TimerNivelHandler,
299 .param_p = NULL
300 };
301 TimerInit(&timer_nivel);
302
303 /* Configurar timer para control de nivel (cada 0,5 s) */
304 timer_config_t timer_control = {
305 .timer = TIMER_C,
306 .period = CONTROL_PERIOD_US,
307 .func_p = TimerControlHandler,
308 .param_p = NULL
309 };
310 TimerInit(&timer_control);
311
312
313 /* Crear tareas */
314 xTaskCreate(&MedirNivelTask, "NivelTask", 512, NULL, 5, &medir_task_handle);
315 xTaskCreate(&ControlNivelTask, "ControlTask", 512, NULL, 5, &control_task_handle);
316 xTaskCreate(&ControlManualTask, "ControlManualTask", 512, NULL, 5, &control_manual_task_handle);
317
318
319 /* Iniciar timer */
320 TimerStart(timer_nivel.timer);
321 TimerStart(timer_control.timer);
322}
